void main() {  List<int> numbers = [5, 12, 3, 8, 6, 13, 4];  List<int> sortedEvenNumbers = getSortedEvenNumbers(numbers);  print(sortedEvenNumbers);List<int> getSortedEvenNumbers(List<int> numbers) {  List<int> evenNumbers = [];  // Filter even numbers manually  for (int number in numbers) {    if (number % 2 == 0) {      evenNumbers.add(number);    }  }  // Sort the even numbers using bubble sort  int n = evenNumbers.length;  for (int i = 0; i < n - 1; i++) {    for (int j = 0; j < n - i - 1; j++) {      if (evenNumbers[j] > evenNumbers[j + 1]) {        // Swap evenNumbers[j] and evenNumbers[j+1]        int temp = evenNumbers[j];        evenNumbers[j] = evenNumbers[j + 1];        evenNumbers[j + 1] = temp;      }    }  }  return evenNumbers;}
